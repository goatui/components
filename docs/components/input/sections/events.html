On input change a CustomEvent goat:change is triggered by the element.

{% highlight js %} document.querySelector('#name-input-field').addEventListener('goat:input', function(event) { console.log('goat:input :: ' + event.target.value); }); document.querySelector('#name-input-field').addEventListener('goat:change', function(event) { console.warn('goat:change :: ' + event.target.value); }); {% endhighlight %}
